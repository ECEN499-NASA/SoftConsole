820 /*------------------------------------------------------------------------------
821  CoreI2C interrupt service routine.
822  ------------------------------------------------------------------------------
823  The function I2C_isr is the CoreI2C interrupt service routine. User must
824  call this function from their application level CoreI2C interrupt handler
825  function. This function runs the I2C state machine based on previous and
826  current status.
827  ------------------------------------------------------------------------------
828  @param this_i2c:
829  The this_i2c parameter is a pointer to the i2c_instance_t data structure
830  holding all data related to a specific CoreI2C channel. For example, if only
831  one channel is initialized, this data structure holds the information of
832  channel 0 of the instantiated CoreI2C hardware.
833 
834  @return none
835 
836  Example:
837  @code
838 
839  #define COREI2C_BASE_ADDR 0xC0000000u
840  #define COREINTERRUPT_BASE_ADDR 0xCC000000u
841  #define COREI2C_SER_ADDR 0x10u
842  #define I2C_IRQ_NB 2u
843 
844  i2c_instance_t g_i2c_inst;
845 
846  void core_i2c_isr( void )
847  {
848  I2C_isr( &g_i2c_inst );
849  }
850 
851  void main( void )
852  {
853  CIC_init( COREINTERRUPT_BASE_ADDR );
854  NVIC_init();
855  CIC_set_irq_handler( I2C_IRQ_NB, core_i2c_isr );
856  I2C_init( &g_i2c_inst, COREI2C_BASE_ADDR, COREI2C_SER_ADDR,
857  I2C_PCLK_DIV_256 );
858  NVIC_enable_interrupt( NVIC_IRQ_0 );
859  }
860  @endcode
861  */
